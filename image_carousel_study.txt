class _HomeScreenState extends State<HomeScreen> {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: PageView(children: [
        Image.asset('asset/image/image_1.jpg', fit: BoxFit.cover,),
        Image.asset('asset/image/image_2.jpg', fit: BoxFit.cover),
        Image.asset('asset/image/image_3.jpg', fit: BoxFit.cover),
        Image.asset('asset/image/image_4.jpg', fit: BoxFit.cover),
        Image.asset('asset/image/image_5.jpg', fit: BoxFit.cover),
      ]),
    );
  }
}

class _HomeScreenState extends State<HomeScreen> {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: PageView(
        children: [1, 2, 3, 4, 5].map((e) => Image.asset('asset/image/image_$e.jpg', fit: BoxFit.cover,)).toList()
      ),
    );
  }
}
